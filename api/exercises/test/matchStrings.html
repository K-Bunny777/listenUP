Levenshtein Distance

String.levenshtein(myString) - calculate Levenshtein distance between two strings


var words = ["rodion","dunia","raskolnikov","adrian","statescu",
             "sunday","saturday","jquery","mootools","dojo"];

function matchWords(input) {

            return words.filter(function(word){

              //compute levensthein distance
              var lev = word.levenshtein(input);

              if(lev >=0 && lev <= 3 ) {

                  return word;
              }

            });
}

Example => Enter: rascolnicov
                  (approximate string match: 'raskolnikov')
                  sandei
                  (approximate string match: 'sunday')
